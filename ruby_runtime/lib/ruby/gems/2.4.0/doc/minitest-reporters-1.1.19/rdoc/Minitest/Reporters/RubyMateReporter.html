def record(test)
  super
  if test.skipped?
    print 'SKIP'
    print_test_with_time(test)
    puts
    puts
  elsif test.error?
    print 'ERROR'
    print_test_with_time(test)
    puts
    print_info(test.failure)
    puts
  elsif test.failure
    print 'FAIL'
    print_test_with_time(test)
    puts
    print_info(test.failure, false)
    puts
  end
end

def print_test_with_time(test)
  print(" #{test.class}##{test.name} (%.2fs)" % test.time)
end
