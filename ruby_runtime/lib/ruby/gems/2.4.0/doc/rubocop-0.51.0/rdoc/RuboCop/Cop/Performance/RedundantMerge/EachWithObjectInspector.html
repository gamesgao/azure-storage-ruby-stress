def value_used?

  receiver.loc.name.source == second_argument.loc.name.source
end

def second_argument
  parent = node.parent
  parent = parent.parent if parent.begin_type?

  @second_argument ||= each_with_object_node(parent)
end

def unwind(receiver)
  while receiver.respond_to?(:send_type?) && receiver.send_type?
    receiver, = *receiver
  end
  receiver
end
