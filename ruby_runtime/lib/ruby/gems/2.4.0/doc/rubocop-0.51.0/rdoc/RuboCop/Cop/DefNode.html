def non_public?(node)
  non_public_modifier?(node.parent) ||
    preceding_non_public_modifier?(node)
end

def preceding_non_public_modifier?(node)
  stripped_source_upto(node.loc.line).any? do |line|
    NON_PUBLIC_MODIFIERS.include?(line)
  end
end
