    class SimpleCov::LinesClassifier

    Classifies whether lines are relevant for code coverage analysis. Comments & whitespace lines, and :nocov: token blocks, are considered not relevant.

def self.no_cov_line

def classify(lines)
  skipping = false

  lines.map do |line|
    if line =~ self.class.no_cov_line
      skipping = !skipping
      NOT_RELEVANT
    elsif skipping || line =~ WHITESPACE_OR_COMMENT_LINE
      NOT_RELEVANT
    else
      RELEVANT
    end
  end
end
