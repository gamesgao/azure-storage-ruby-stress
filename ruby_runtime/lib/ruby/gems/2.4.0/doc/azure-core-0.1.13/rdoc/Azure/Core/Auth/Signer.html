def initialize(access_key)
  if access_key.nil?
    raise ArgumentError, 'Signing key must be provided'
  end

  @access_key = Base64.strict_decode64(access_key)
end

def sign(body)
  signed = OpenSSL::HMAC.digest('sha256', access_key, body)
  Base64.strict_encode64(signed)
end
